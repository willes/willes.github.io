 <div>
  父组件:{{age}}
  <Son :data="age"></Son>
 </div>
</template>

import Son from "./Son";
export default {
 components: {
  Son
 },

# 2. $refs传递数据

父组件通过$refs来获取子组件的方法和属性

<template>
    父组件:
    <Son ref="son"/>
</template>
<script>
  export default {
      mounted () {
          this.$refs.son.test()
      }
  }
</script>

# 3. $emit使用

子组件触发父组件方法,通过回调的方式将修改的内容传递给父组件

<template>
 <div>
  父组件:{{age}}
  <Son :data="age" @input="change"></Son>
 </div>
</template>
<script>
import Son from "./Son";
export default {
 data() {
  return { age: 100 };
 }
 methods: {
  change(sonData) {
   this.age = sonData;
  }
 },
 components: {
  Son
 }
};
</script>
